export const getComments = async (storyId, limit = 50, offset = 0) => {
  const response = await api.get(`/stories/${storyId}/comments`, {
    params: { limit, offset }
  })
  return response.data
}

// 发表评论
export const createComment = async (storyId, commentData) => {
  const response = await api.post(`/stories/${storyId}/comments`, commentData)
  return response.data
}

// ========== 点赞相关 ==========

// 获取点赞信息
export const getLikes = async (storyId, userId = 'anonymous') => {
  const response = await api.get(`/stories/${storyId}/likes`, {
    params: { user_id: userId }
  })
  return response.data
}

// 点赞故事
export const likeStory = async (storyId, userIdentifier = 'anonymous') => {
  const response = await api.post(`/stories/${storyId}/like`, {
    user_identifier: userIdentifier
  })
  return response.data
}

// 取消点赞
export const unlikeStory = async (storyId, userIdentifier = 'anonymous') => {
  const response = await api.delete(`/stories/${storyId}/like`, {
    data: { user_identifier: userIdentifier }
  })
  return response.data
}

// ========== 故事列表增强 ==========

// 获取故事列表（支持筛选和排序）
export const getStoriesAdvanced = async (params = {}) => {
  const response = await api.get('/stories', { params })
  return response.data
}

// 搜索故事
export const searchStories = async (query, limit = 20) => {
  const response = await api.get('/stories/search', {
    params: { q: query, limit }
  })
  return response.data
}

#### 1.1 CommentList.jsx

```jsx
import { useState, useEffect } from 'react'
import { getComments, createComment } from '../api/api'

export default function CommentList({ storyId }) {
  const [comments, setComments] = useState([])
  const [total, setTotal] = useState(0)
  const [loading, setLoading] = useState(true)
  const [submitting, setSubmitting] = useState(false)
  
  // 表单状态
  const [author, setAuthor] = useState('')
  const [content, setContent] = useState('')

  // 加载评论
  useEffect(() => {
    loadComments()
  }, [storyId])

  const loadComments = async () => {
    try {
      setLoading(true)
      const data = await getComments(storyId)
      setComments(data.comments)
      setTotal(data.total)
    } catch (error) {
      console.error('加载评论失败:', error)
      alert('加载评论失败，请刷新重试')
    } finally {
      setLoading(false)
    }
  }

  // 提交评论
  const handleSubmit = async (e) => {
    e.preventDefault()
    
    if (!author.trim()) {
      alert('请输入您的昵称')
      return
    }
    
    if (!content.trim()) {
      alert('请输入评论内容')
      return
    }
    
    try {
      setSubmitting(true)
      await createComment(storyId, { author, content })
      
      // 清空表单
      setAuthor('')
      setContent('')
      
      // 重新加载评论
      await loadComments()
      
      alert('评论发表成功！')
    } catch (error) {
      console.error('发表评论失败:', error)
      alert('发表评论失败，请重试')
    } finally {
      setSubmitting(false)
    }
  }

  if (loading) {
    return (
      <div className="animate-pulse space-y-4">
        <div className="h-20 bg-gray-200 rounded"></div>
        <div className="h-20 bg-gray-200 rounded"></div>
      </div>
    )
  }

  return (
    <div className="space-y-6">
      {/* 评论统计 */}
      <h3 className="text-xl font-bold text-gray-800">
        💬 评论 ({total})
      </h3>

      {/* 评论表单 */}
      <form onSubmit={handleSubmit} className="bg-gray-50 p-4 rounded-lg space-y-3">
        <input
          type="text"
          placeholder="您的昵称"
          value={author}
          onChange={(e) => setAuthor(e.target.value)}
          className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
          maxLength={20}
        />
        
        <textarea
          placeholder="写下您的评论..."
          value={content}
          onChange={(e) => setContent(e.target.value)}
          rows={3}
          className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
          maxLength={500}
        />
        
        <div className="flex justify-between items-center">
          <span className="text-sm text-gray-500">
            {content.length}/500
          </span>
          <button
            type="submit"
            disabled={submitting}
            className="px-6 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 disabled:bg-gray-400 transition-colors"
          >
            {submitting ? '发表中...' : '发表评论'}
          </button>
        </div>
      </form>

      {/* 评论列表 */}
      <div className="space-y-4">
        {comments.length === 0 ? (
          <p className="text-center text-gray-500 py-8">
            还没有评论，快来抢沙发吧！
          </p>
        ) : (
          comments.map((comment) => (
            <div
              key={comment.id}
              className="bg-white p-4 rounded-lg border border-gray-200 hover:shadow-md transition-shadow"
            >
              <div className="flex justify-between items-start mb-2">
                <span className="font-semibold text-gray-800">
                  {comment.author}
                </span>
                <span className="text-sm text-gray-500">
                  {new Date(comment.created_at).toLocaleString('zh-CN')}
                </span>
              </div>
              <p className="text-gray-700 leading-relaxed">
                {comment.content}
              </p>
            </div>
          ))
        )}
      </div>
    </div>
  )
}
```

### 2. 点赞组件

#### 2.1 LikeButton.jsx

在 `frontend/src/components/LikeButton.jsx` 创建：

```jsx
import { useState, useEffect } from 'react'
import { getLikes, likeStory, unlikeStory } from '../api/api'

export default function LikeButton({ storyId }) {
  const [likesCount, setLikesCount] = useState(0)
  const [userLiked, setUserLiked] = useState(false)
  const [loading, setLoading] = useState(false)
  
  // 获取用户标识（简单版本，使用 localStorage）
  const getUserId = () => {
    let userId = localStorage.getItem('user_id')
    if (!userId) {
      userId = 'user_' + Math.random().toString(36).substr(2, 9)
      localStorage.setItem('user_id', userId)
    }
    return userId
  }

  // 加载点赞状态
  useEffect(() => {
    loadLikes()
  }, [storyId])

  const loadLikes = async () => {
    try {
      const userId = getUserId()
      const data = await getLikes(storyId, userId)
      setLikesCount(data.likes_count)
      setUserLiked(data.user_liked)
    } catch (error) {
      console.error('加载点赞状态失败:', error)
    }
  }

  // 处理点赞/取消点赞
  const handleLike = async () => {
    if (loading) return
    
    try {
      setLoading(true)
      const userId = getUserId()
      
      if (userLiked) {
        // 取消点赞
        const data = await unlikeStory(storyId, userId)
        setLikesCount(data.likes_count)
        setUserLiked(false)
      } else {
        // 点赞
        const data = await likeStory(storyId, userId)
        setLikesCount(data.likes_count)
        setUserLiked(true)
      }
    } catch (error) {
      console.error('点赞操作失败:', error)
      alert(error.response?.data?.detail || '操作失败，请重试')
    } finally {
      setLoading(false)
    }
  }

  return (
    <button
      onClick={handleLike}
      disabled={loading}
      className={`flex items-center space-x-2 px-4 py-2 rounded-lg transition-all ${
        userLiked
          ? 'bg-red-100 text-red-600 hover:bg-red-200'
          : 'bg-gray-100 text-gray-600 hover:bg-gray-200'
      } disabled:opacity-50`}
    >
      <span className="text-xl">
        {userLiked ? '❤️' : '🤍'}
      </span>
      <span className="font-semibold">
        {likesCount}
      </span>
    </button>
  )
}
```

### 4. 视频生成进度显示

#### 4.1 VideoStatus.jsx 组件

```jsx
import { useState, useEffect } from 'react'
import { getStory } from '../api/api'

export default function VideoStatus({ storyId, onStatusChange }) {
  const [status, setStatus] = useState('pending')
  const [videoUrl, setVideoUrl] = useState(null)

  useEffect(() => {
    if (status === 'generating') {
      // 每5秒轮询一次状态
      const interval = setInterval(checkStatus, 5000)
      return () => clearInterval(interval)
    }
  }, [status])

  const checkStatus = async () => {
    try {
      const data = await getStory(storyId)
      setStatus(data.video_status)
      setVideoUrl(data.video_url)
      
      if (onStatusChange) {
        onStatusChange(data.video_status, data.video_url)
      }
    } catch (error) {
      console.error('检查视频状态失败:', error)
    }
  }

  const renderStatus = () => {
    switch (status) {
      case 'pending':
        return (
          <div className="text-gray-600">
            ⏸️ 尚未生成视频
          </div>
        )
      
      case 'generating':
        return (
          <div className="flex items-center space-x-3">
            <div className="animate-spin rounded-full h-5 w-5 border-b-2 border-blue-600"></div>
            <span className="text-blue-600 font-medium">
              🎬 视频生成中，请稍候...
            </span>
          </div>
        )
      
      case 'completed':
        return (
          <div className="text-green-600 font-medium">
            ✅ 视频已生成！
          </div>
        )
      
      case 'failed':
        return (
          <div className="text-red-600 font-medium">
            ❌ 视频生成失败
          </div>
        )
      
      default:
        return null
    }
  }

  return (
    <div className="bg-gray-50 p-4 rounded-lg">
      {renderStatus()}
    </div>
  )
}
```
